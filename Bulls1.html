<script>

  let secret = generateSecret();

  const log = document.getElementById("log");

  const title = document.getElementById("title");

  const inputField = document.getElementById("guess");

  const enterBtn = document.getElementById("enterBtn");

  const timerDisplay = document.getElementById("timer");


  let gameOver = false;

  let timer = 0;

  let timerInterval = null;

  let firstGuessMade = false;


  function generateSecret() {

    const digits = [];

    while (digits.length < 4) {

      const d = Math.floor(Math.random() * 10);

      if (!digits.includes(d)) digits.push(d);

    }

    return digits.join("");

  }


  function formatTime(seconds) {

    const min = Math.floor(seconds / 60).toString().padStart(2, '0');

    const sec = (seconds % 60).toString().padStart(2, '0');

    return `${min}:${sec}`;

  }


  function startTimer() {

    if (firstGuessMade) return;

    firstGuessMade = true;

    timerInterval = setInterval(() => {

      timer++;

      timerDisplay.textContent = formatTime(timer);

    }, 1000);

  }


  function showTemporaryMessage(message) {

    const original = "Bulls & Cows";

    title.textContent = message;

    title.style.opacity = 1;

    title.style.color = "red";


    setTimeout(() => {

      let opacity = 1;

      const fade = setInterval(() => {

        opacity -= 0.05;

        if (opacity <= 0) {

          clearInterval(fade);

          title.textContent = original;

          title.style.color = "black";

          title.style.opacity = 1;

        } else {

          title.style.opacity = opacity;

        }

      }, 25);

    }, 4000);

  }


  function checkGuess() {

    if (gameOver) return;


    startTimer();


    const guess = inputField.value;


    if (!/^\d{4}$/.test(guess) || new Set(guess).size !== 4) {

      showTemporaryMessage("4 Unique Digits");

      return;

    }


    let bulls = 0;

    let cows = 0;


    for (let i = 0; i < 4; i++) {

      if (guess[i] === secret[i]) bulls++;

      else if (secret.includes(guess[i])) cows++;

    }


    log.innerHTML += `

      <div class="row">

        <div>${guess}</div>

        <div class="bull">${bulls}</div>

        <div class="cow">${cows}</div>

      </div>

    `;


    if (bulls === 4) {

      log.innerHTML += `<div class="win">YOU WIN</div>`;

      gameOver = true;

      inputField.disabled = true;

      clearInterval(timerInterval);

      enterBtn.textContent = "Restart";

      enterBtn.onclick = restartGame;

    }


    inputField.value = "";

    inputField.focus();

  }


  function restartGame() {

    secret = generateSecret();

    // Remove all guess rows but keep the header

    log.querySelectorAll(".row, .win").forEach(el => el.remove());


    inputField.disabled = false;

    inputField.value = "";

    inputField.focus();

    gameOver = false;

    firstGuessMade = false;

    timer = 0;

    timerDisplay.textContent = "00:00";

    enterBtn.textContent = "Enter";

    enterBtn.onclick = checkGuess;

    if (timerInterval) clearInterval(timerInterval);

  }


  inputField.addEventListener("keydown", function(e) {

    if (e.key === "Enter") {

      checkGuess();

    }

  });

</script>
